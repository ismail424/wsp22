def self.filter_by_genre(genre)
    results = []

    db.execute("SELECT * FROM #{table_name} WHERE genres LIKE ?", "%#{genre}%").each do |row|
        results << self.new(row)
    end

    results
end

        result << Genre.find(row['genre_id'])
    end
    result
